Function operator<<

boost::svg::operator<<

Synopsis

// In header: <boost/svg_plot/svg_style.hpp>


std::ostream & operator<<(std::ostream & os, const text_style & ts);

Description

Output a text_style as a text string (mainly useful for diagnostic use).

Example: text_style ts(12, "Arial", "italic", "bold", "", "", 0); std::cout << ts << std::endl; Outputs: text_style(12, "Arial", "italic", "bold", "", "")

text_style ts(12, "Arial", "italic", "bold", "narrower", "underline", 1000); std::cout << ts << std::endl; Outputs: text_style(12, "Arial", "italic", "bold", "narrower", "underline", 1000)
